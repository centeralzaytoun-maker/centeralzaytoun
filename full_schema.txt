create table public.universal_inbox (
  id uuid not null default extensions.uuid_generate_v4 (),
  recipient_role text null,
  recipient_id uuid null,
  title text not null,
  body text not null,
  category text null,
  priority text null default 'normal'::text,
  action_type text null,
  action_id uuid null,
  status text null default 'open'::text,
  assigned_to uuid null,
  internal_note text null,
  read_at timestamp with time zone null,
  created_at timestamp with time zone null default now(),
  center_id uuid null,
  constraint universal_inbox_pkey primary key (id),
  constraint universal_inbox_assigned_to_fkey foreign KEY (assigned_to) references auth.users (id),
  constraint universal_inbox_category_check check (
    (
      category = any (
        array[
          'attendance'::text,
          'exam'::text,
          'payment'::text,
          'system'::text,
          'request'::text
        ]
      )
    )
  ),
  constraint universal_inbox_priority_check check (
    (
      priority = any (
        array['normal'::text, 'important'::text, 'urgent'::text]
      )
    )
  ),
  constraint universal_inbox_recipient_role_check check (
    (
      recipient_role = any (
        array['student'::text, 'admin'::text, 'staff'::text]
      )
    )
  ),
  constraint universal_inbox_status_check check (
    (
      status = any (
        array[
          'open'::text,
          'processing'::text,
          'resolved'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;

create trigger set_center_id_on_inbox_insert BEFORE INSERT on universal_inbox for EACH row
execute FUNCTION set_inbox_center_id ();

create trigger tr_set_center_id_universal_inbox BEFORE INSERT on universal_inbox for EACH row
execute FUNCTION universal_set_center_id ();


create table public.wallet_transactions (
  id uuid not null default extensions.uuid_generate_v4 (),
  student_id uuid null,
  amount double precision not null,
  type text not null,
  description text null,
  created_at timestamp with time zone not null default now(),
  created_by uuid null,
  balance_after numeric null default 0,
  notes text null,
  center_id uuid null,
  constraint wallet_transactions_pkey primary key (id),
  constraint wallet_transactions_created_by_fkey foreign KEY (created_by) references staff_profiles (id) on delete set null,
  constraint wallet_transactions_student_id_fkey foreign KEY (student_id) references students (id) on delete CASCADE
) TABLESPACE pg_default;

create index IF not exists idx_wallet_center on public.wallet_transactions using btree (center_id) TABLESPACE pg_default;

create index IF not exists idx_wallet_transactions_center_id on public.wallet_transactions using btree (center_id) TABLESPACE pg_default;

create trigger set_center_id_on_wallet_insert BEFORE INSERT on wallet_transactions for EACH row
execute FUNCTION set_wallet_center_id ();

create trigger tr_set_center_id_wallet_transactions BEFORE INSERT on wallet_transactions for EACH row
execute FUNCTION universal_set_center_id ();
